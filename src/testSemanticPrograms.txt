  x = 0
  print(x)
}$

/* Test 2: Undeclared variable - y was never declared */
{
  int x
  x = 5
  y = 3
}$

{
  string s
  s = 5
}$

/* Test 4: Type mismatch - assigning string to int variable */
{
  int x
  x = "hello"
}$

/* Test 5: Type mismatch - assigning int to boolean variable */
{
  boolean b
  b = 9
}$

/* Test 6: Unused variable warning - f is declared but never used */
{
  int x
  x = 1
  print(x)
  boolean f
}$

/* Test 7: Uninitialized variable warning - printing x before assigning */
{
  int x
  print(x)
}$

/* Test 8: Nested scope - inner scope can see parent variable */
{
  int x
  x = 5
  {
    print(x)
  }
}$

/* Test 9: Sibling scope isolation - second block should NOT see first block's variable */
{
  {
    int x
    x = 1
  }
  {
    print(x)
  }
}$

/* Test 10: Valid if with boolval */
{
  int x
  x = 1
  if true {
    print(x)
  }
}$

/* Test 11: Valid while with boolean condition - types match */
{
  int x
  x = 0
  while (x == 1) {
    print(x)
  }
}$

{
  int x
  x = 0
  string s
  s = "a"
  while (x == s) {
    print(x)
  }
}$
